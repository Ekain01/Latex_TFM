\section{Código WinCC}
En esta sección se presentan los algoritmos de filtrado implementados mediante scripts globales en WinCC. El código se ha desarrollado en lenguajes de programación compatibles con WinCC, como VBScript o C, y se ha incluido con comentarios explicativos para facilitar su comprensión. Estos algoritmos se encargan de filtrar y organizar la visualización de los aerogeneradores en función de su estado operativo y su pertenencia a parques específicos, además de gestionar la navegación entre diferentes pantallas del sistema SCADA.
\subsection{\texttt{AplicarFiltroPosicion()}}
\begin{lstlisting}
void AplicarFiltroPosicion(){

    int selValue, parkValue;
    int CANT_MOLINOS, i;
    int tagRun, tagEmg, tagPark, state;
    int match;

    // Variables de cuadricula
    int startX = 10;
    int startY = 0;
    int offsetX = 475;
    int offsetY = 220;
    int maxCols = 4;

    int currentX;
    int currentY;
    int colCount = 0;
    //Nombres de tags
    char tagNameRun[100];
    char tagNameEmg[100];
    char tagNamePark[100];
    char FP[50];

    // Guarda IDs de molinos que cumplen el filtro
    int validList[120];
    int validCount = 0;

    char* prefijoActual;

    currentX = startX;
    currentY = startY;
    // Obtener valores de filtros
    selValue = GetTagWord("HMI_Filtro_Molinos");
    parkValue = GetTagWord("HMI_Filtro_Parques");
    prefijoActual = "WTG";
    CANT_MOLINOS = 120;
    // Evaluar cada molino
    for (i = 1; i <= CANT_MOLINOS; i++) {
        sprintf(tagNameEmg, "%s%d.WTUR.TurSt.stVal", prefijoActual, i);
        sprintf(tagNamePark, "%s%d.LLNO.InSSet.setSrc", prefijoActual, i);

        tagEmg = GetTagWord(tagNameEmg);
        tagPark = GetTagWord(tagNamePark);

        switch (tagEmg) {
            case 0:
                state = 2;
                break;
            case 1:
            case 4:
                state = 3;
                break;
            case 2:
                state = 4;
                break;
            case 3:
                state = 5;
                break;
            case 5:
            case 7:
                state = 6;
                break;
            case 6:
                state = 7;
                break;
            default:
                if (tagEmg >= 10 && tagEmg <= 19) {
                    state = 8;
                } else {
                    state = -1;
                }
                break;
        }

        match = 0;
        if (parkValue == 1 || parkValue == tagPark) {
            if (selValue == 1 || selValue == state) {
                match = 1;
            }
        }

        if (match == 1) {
            validList[validCount] = i;
            validCount++;
        }
    }
    // Visualizar molinos que cumplen el filtro
    for (i = 1; i <= CANT_MOLINOS; i++) {
        sprintf(FP, "PW_%d", i);
        SetPropBOOL("20_WTG_Turbine_Overview.PDL", FP, "Visible", FALSE);
    }

    for (i = 0; i < validCount; i++) {
        int molinoID = validList[i];
        sprintf(FP, "PW_%d", molinoID);

        SetPropBOOL("20_WTG_Turbine_Overview.PDL", FP, "Visible", TRUE);
        SetPropDouble("20_WTG_Turbine_Overview.PDL", FP, "Left", currentX);
        SetPropDouble("20_WTG_Turbine_Overview.PDL", FP, "Top", currentY);

        colCount++;
        if (colCount >= maxCols) {
            currentX = startX;
            currentY += offsetY;
            colCount = 0;
        } else {
            currentX += offsetX;
        }
    }
}

#include "apdefap.h" 
void AplicarFiltroParque(){ 

    //Variables
   
    int selValue, parkValue; 
    int CANT_VENTANAS = 22; 
    int i, k; 
    int match; 

    int tagParkBase;    // Parque del molino principal (TagPrefix)
    int currentPark;    // Parque del molino que estamos evaluando en el bucle k
    int baseID;         // ID numerico del molino principal 
    int currentID;      
    int rawStatus; 
    int state; 

    int startX = 10;    
    int startY = 0;     
    int offsetX = 475;  
    int offsetY = 225;  
    int maxCols = 4; 
    
    int currentX = startX; 
    int currentY = startY; 
    int colCount = 0;  

    char FP[50];  
    char* prefijoVentana; 
    char tagName[150]; 

    //Leer valores de filtros
    selValue = GetTagWord("HMI_Filtro_Molinos"); 
    parkValue = GetTagWord("HMI_Filtro_Parques"); 

  
    // Evaluar cada ventana de imagen (cada una con un parque base diferente)

    for (i = 1; i <= CANT_VENTANAS; i++) { 
        match = 0; 
        
        sprintf(FP, "PK_%d", i); 
        
        // LEEMOS EL TAG PREFIX
        prefijoVentana = GetPropChar("21_WTG_Park_Overview.PDL", FP, "TagPrefix"); 
        
        if (prefijoVentana != NULL && strlen(prefijoVentana) > 0) { 
            
            // Leemos el parque del molino base para referencia
            sprintf(tagName, "%s.LLNO.SubSt.setSrc", prefijoVentana); 
            tagParkBase = GetTagWord(tagName); 
            
            // Leemos el ID numerico base
            sprintf(tagName, "%s.LLNO.UnitID.setSrc", prefijoVentana); 
            baseID = GetTagWord(tagName); 

            if (baseID > 0) {
                // BUCLE K: Comprobamos el molino base (0), el +1 y el +2
                for (k = 0; k < 3; k++) { 
                    currentID = baseID + k; 

                    // 1. Obtener parque del molino actual (k) para ver si esta en el parque filtrado
                    if (k == 0) {
                        currentPark = tagParkBase;
                    } else {
                        sprintf(tagName, "WTG%d.LLNO.SubSt.setSrc", currentID); 
                        currentPark = GetTagWord(tagName);
                    }

                    // FILTRO DE PARQUE: El molino 'k' debe pertenecer al parque seleccionado
                    if (parkValue == 1 || parkValue == currentPark) {
                        
                        // 2. Leer estado del molino 'k'
                        if (k == 0) {
                            sprintf(tagName, "%s.WTUR.TurSt.stVal", prefijoVentana);
                        } else {
                            sprintf(tagName, "WTG%d.WTUR.TurSt.stVal", currentID);
                        }
                        rawStatus = GetTagWord(tagName); 

                        // 3. Maquina de estados interna 
                        switch (rawStatus) { 
                            case 0: state = 2; break; // Stopped 
                            case 1: 
                            case 4: state = 3; break; // Starting o Shutting down 
                            case 2: state = 4; break; // Running 
                            case 3: state = 5; break; // Paused 
                            case 5: 
                            case 7: state = 6; break; // Emergency stop o Fault 
                            case 6: state = 7; break; // Maintenance 
                            default: 
                                if (rawStatus >= 10 && rawStatus <= 19) { 
                                    state = 8; // Communication Error 
                                } else { 
                                    state = -1; 
                                } 
                                break; 
                        } 

                        //filtro estado
                        if (selValue == 1 || selValue == state) { 
                            match = 1;  
                            break; // Si uno de los tres cumple, ya no hace falta mirar mas
                        } 
                    }
                } 
            } 
        } 
        
       
        // Visualizacion de ventanas: Si 'match' es 1, esta ventana se muestra en la cuadricula, si no se oculta
    
        if (match == 1) { 
            SetPropDouble("21_WTG_Park_Overview.PDL", FP, "Left", currentX); 
            SetPropDouble("21_WTG_Park_Overview.PDL", FP, "Top", currentY);  
            SetPropBOOL("21_WTG_Park_Overview.PDL", FP, "Visible", TRUE); 
            
            colCount++; 
            if (colCount >= maxCols) { 
                currentX = startX; 
                currentY += offsetY; 
                colCount = 0; 
            } else { 
                currentX += offsetX; 
            } 
        } else { 
            SetPropBOOL("21_WTG_Park_Overview.PDL", FP, "Visible", FALSE); 
        } 
    } 
}

\section{Código NodeRED}

En esta sección se presenta el código de las funciones más relevantes implementadas en NodeRED para la simulación y control del aerogenerador. Estos fragmentos de código corresponden a las funciones JavaScript utilizadas en los nodos de función dentro de los flujos de NodeRED, y se han incluido con comentarios explicativos para facilitar su comprensión.
\subsection{Nodo : parametro de viento}
Este nodo se encarga de calcular el viento utilizado para los cálculos analógicos a partir de la consigna del usuario. De esta forma, cuando el usuario cambia la consigna de viento, el valor virtual simulado aumenta o disminuye progresivamente. Además, se implementa una lógica de emergencia que hace que el viento virtal caiga a 0 si se supera un umbral de velocidad o si se activa una alarma o mantenimiento.
\begin{lstlisting}
    // 1. Intentar leer los valores
let windSpd = flow.get("windSpd") || 0; // Viento real del sensor
let target = flow.get("targetWind") || 0;
let current = flow.get("windSpdTurbine") || 0;
let alarm = flow.get("Alarmas") || 0;
let mantenimiento = flow.get("Mantenimiento")|| 0 ;
// 3. Configuracion de la rampa
let paso = 0.5; 
// 2. Logica de EMERGENCIA
// Si el viento supera los 25 m/s, el objetivo real de la turbina DEBE ser 0
if (windSpd > 25) {
    target = 0;
    paso=1;
}
if(alarm){
    target = 0;
    paso=1;
}
if(mantenimiento){
    target = 0;
    paso=1;
}
// Asegurarnos de que sean numeros
target = parseFloat(target);
current = parseFloat(current);



// 4. Logica de movimiento (Rampa)
if (Math.abs(current - target) < paso) {
    current = target;
} else if (current < target) {
    current += paso;
} else if (current > target) {
    current -= paso;
}

// 5. Guardar el resultado para el proximo ciclo
flow.set("windSpdTurbine", current);

// 6. Enviar el valor al SCADA/Kepware
msg.payload = current;

return msg;
\end{lstlisting}
\subsection{Nodo: Potencia generada}
Este nodo se encarga de calcular la potencia generada por el aerogenerador en función de la velocidad del viento virtual. Se implementa una lógica basada en la curva de potencia típica de un aerogenerador, que incluye una fase de arranque, una fase de potencia creciente, una fase de potencia nominal y una fase de corte por viento excesivo.
\begin{lstlisting}
    // 1. Recoger velocidad del slider
let v = parseFloat(flow.get("windSpdTurbine")) || 0;

// Parametros de la curva (2.5 MW)
const cutIn = 3.5;
const nominal = 13.0;
const cutOut = 25.0;
const potMax = 2500.0;

let potencia = 0;

// 2. Logica de la curva de potencia
if (v >= cutIn && v < nominal) {
    // Curva cubica entre arranque y nominal
    potencia = potMax * Math.pow((v - cutIn) / (nominal - cutIn), 3);
} else if (v >= nominal && v <= cutOut) {
    // Potencia maxima constante
    potencia = potMax;
} else {
    // Por debajo de cut-in o por encima de cut-out: 0
    potencia = 0;
}

// Redondeamos a 2 decimales
potencia = Math.round(potencia * 100) / 100;

// 3. Enviar el valor
msg.payload = potencia;

return msg;
\end{lstlisting}
\subsection{Nodo: Calculo viento actual}
Este nodo se encarga de calcular el valor de viento actual que se enviará a la variable dentro de WMET para que salga visualizandose en el SCADA. El valor es el mismo que el del viento actual salvo cuando entra en modo de emergencia, pausa o mantenimiento.
\begin{lstlisting}
    // 1. Intentar leer los valores
let target = flow.get("targetWind");
let current = flow.get("windSpd");

// 2. Si es la primera vez que se ejecuta, forzamos valores iniciales
if (target === undefined) target = 0;
if (current === undefined) current = 0;

// Asegurarnos de que sean numeros
target = parseFloat(target);
current = parseFloat(current);

// 3. Configuracion de la rampa
let paso = 0.5; 

// 4. Logica de movimiento
if (Math.abs(current - target) < paso) {
    current = target;
} else if (current < target) {
    current += paso;
} else if (current > target) {
    current -= paso;
}

// 5. GUARDAR el resultado
flow.set("windSpd", current);


// 5. Enviar el valor 
msg.payload=current;

return msg;
\end{lstlisting}
\subsection{Nodo :  Orientación}
Este nodo se encarga de calcular el valor de orientación de la turbina, calculando la dirección de giro en base a la posición actual y la posición objetivo, y actualizando la posición actual progresivamente para simular el movimiento de la turbina hacia la consigna de orientación establecida por el usuario.
\begin{lstlisting}
    // 1. Obtener la consigna (Target) y la posicion actual (Current)
let target = parseFloat(flow.get("targetAngle")) || 0;
let current = context.get("currentPos") || 0;

// 2. Configurar la velocidad del movimiento
// Cuanto mas alto sea este numero, mas rapido llegara a la consigna
let paso = 0.7; 

// 3. Logica de aproximacion
if (current < target) {
    // Si la posicion actual es menor, subimos
    current += paso;
    // Si nos pasamos del objetivo por el tamano del paso, igualamos
    if (current > target) current = target;
    
} else if (current > target) {
    // Si la posicion actual es mayor, bajamos
    current -= paso;
    // Si nos pasamos bajando, igualamos
    if (current < target) current = target;
}

// 4. Guardar la nueva posicion en la memoria 
context.set("currentPos", current);

// 5. Enviar el valor 
msg.payload=current;
return msg;
\end{lstlisting}
\subsection{Nodo: Orientación de pala}
Este nodo se encarga de calcular el ángulo de pitch de las palas del aerogenerador en función de la velocidad del viento. Se implementa una lógica que ajusta el ángulo de pitch para optimizar la producción de energía y proteger el aerogenerador en condiciones de viento excesivo, siguiendo una curva típica de control de pitch.
\begin{lstlisting}
    // 1. Leer el ultimo valor guardado por el slider en el flujo (flow)
// Si no existe aun (al arrancar), usamos 0 por defecto
let windSpd = flow.get("windSpdTurbine") || 0;

// 2. Parametros de la turbina
const V_ARRANQUE = 3.0;
const V_NOMINAL = 12.0;
const V_CORTE = 25.0;

let pitchAngle = 0;

// 3. Logica de Pitch
if (windSpd < V_ARRANQUE) {
    pitchAngle = 90; 
} else if (windSpd >= V_ARRANQUE && windSpd <= V_NOMINAL) {
    pitchAngle = 0; 
} else if (windSpd > V_NOMINAL && windSpd <= V_CORTE) {
    pitchAngle = ((windSpd - V_NOMINAL) / (V_CORTE - V_NOMINAL)) * 90;
} else {
    pitchAngle = 90; 
}
//4. Envio
msg.payload = pitchAngle;
return msg;
\end{lstlisting}
\subsection{Nodo: Azimut del rotor}
Este nodo se encarga de calcular el ángulo de azimut del rotor del aerogenerador en función de las RPM del rotor. Se implementa una lógica que actualiza el ángulo de azimut progresivamente en función de las RPM, simulando el movimiento continuo del rotor a medida que gira.
\begin{lstlisting}
    // 1. Recuperar las RPM del rotor (calculadas en el nodo anterior)
// Si no hay valor, asumimos 0 RPM
let rpm = parseFloat(flow.get("rotorRPM")) || 0;

    // 2. Recuperar el angulo actual de la memoria del contexto
let currentAngle = context.get("currentAngle") || 0;

// 3. Logica de conversion: de RPM a Grados por segundo
// Una revolucion = 360 grados.
// RPM / 60 = Revoluciones por segundo.
// (RPM / 60) * 360 = Grados por segundo.
// Simplificado: RPM * 6.
let gradosPorSegundo = rpm * 6;

// 4. Calcular el incremento para este ciclo de 100ms (0.1 segundos)
let incremento = gradosPorSegundo * 0.1;

// 5. Sumar el incremento al angulo actual
currentAngle += incremento;

// 6. Si pasa de 360 grados, resetear (vuelta a empezar)
// Usamos el operador modulo para mantener precision
if (currentAngle >= 360) {
    currentAngle = currentAngle % 360;
}

// 7. Guardar el nuevo angulo en la memoria del contexto para el proximo ciclo
context.set("currentAngle", currentAngle);

// 8. Enviar el resultado al SCADA/Kepware
msg.payload = parseFloat(currentAngle.toFixed(2));

return msg;
\end{lstlisting}
\subsection{Nodo: Velocidad del rotor}
Este nodo se encarga de calcular las RPM del rotor del aerogenerador en función de la velocidad del viento.
\begin{lstlisting}
    // 1. Recuperar el viento de la memoria (0 si no hay nada)
let windSpd = parseFloat(flow.get("windSpdTurbine")) || 0;
/**
 * Calculo de RPM basado en la velocidad del viento (windSpd)
 * Parametros configurables:
 */
const cutIn = 3;      // Velocidad minima para empezar a girar (m/s)
const cutOut = 25;    // Velocidad de parada por seguridad (m/s)
const ratedSpd = 12;  // Velocidad a la que alcanza RPM maximas (m/s)
const maxRPM = 15;    // RPM maximas del rotor

let rpm = 0;

// 1. LLogica de estados segun velocidad del viento
if (windSpd < cutIn) {
    // Viento insuficiente: El rotor esta en PAUSA o PARADO
    rpm = 0;
} 
else if (windSpd >= cutIn && windSpd < ratedSpd) {
    // Zona de produccion: Las RPM suben linealmente con el viento
    // Formula: rpm = (viento - min) * (maxRPM / (viento_nominal - min))
    rpm = (windSpd - cutIn) * (maxRPM / (ratedSpd - cutIn));
} 
else if (windSpd >= ratedSpd && windSpd <= cutOut) {
    // Zona nominal: El sistema de "pitch" mantiene las RPM constantes
    rpm = maxRPM;
} 
else {
    // EMERGENCIA: Viento > 25 m/s, el rotor se frena a 0
    rpm = 0;
}
flow.set("rotorRPM", msg.payload);

return msg;
\end{lstlisting}
